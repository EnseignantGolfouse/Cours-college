\begin{question}[(4 points)]
	Donne l'abscisse des points A, B, C et D : \vspace{1em}

	\begin{center}
		\begin{tikzpicture}
			\draw[\myarrow] (-5.5,0) -- (5.5,0);
			\foreach \x in {-5,...,5} {
					\draw (\x,0) -- (\x,-0.2);
				}
			\node[below] at (0,-0.2) {0};
			\node[below] at (1,-0.2) {1};

			\coordinate (A) at (2,0);
			\coordinate (B) at (-2,0);
			\coordinate (C) at (-3.5,0);
			\coordinate (D) at (0.5,0);

			\foreach \p in {A,B,C,D} {
					\node at (\p) {×};
					\node[above] at (\p) {\p};
				}
		\end{tikzpicture}
	\end{center} \vspace{1em}

	\dotfill

	Donne l'abscisse des points E, F, G et H : \vspace{1em}

	\begin{center}
		\begin{tikzpicture}[scale=6]
			\draw[\myarrow] (-1.1,0) -- (1.1,0);
			\foreach \x in {-5,...,5} {
					\draw (\x / 5,0) -- (\x / 5,-0.04);
				}
			\node[below] at (-0.4,-0.04) {-0,4};
			\node[below] at (-0.6,-0.04) {-0,6};

			\coordinate (E) at (0.7,0);
			\coordinate (F) at (-0.2,0);
			\coordinate (G) at (-0.4,0);
			\coordinate (H) at (-1,0);

			\foreach \p in {E,F,G,H} {
					\node at (\p) {×};
					\node[above] at (\p) {\p};
				}
		\end{tikzpicture}
	\end{center} \vspace{1em}

	\dotfill
\end{question}

\begin{question}[(3 points)]
	Complète par $<$ ou $>$ :

	\begin{multicols}{3}
		\begin{enumerate}[a)]
			\item 5 ...... 3 % >
			\item -5 ...... -3 % <
			\item -8,51 ...... -8,5 % <
			\item -7,12 ...... 0,8 % <
			\item 3,14 ...... 2,718 % >
			\item -0,99 ...... -0,909 % <
		\end{enumerate}
	\end{multicols}
\end{question}

\begin{question}[(4 points)]
	\begin{enumerate}[1)]
		\item Range les nombres suivants dans l'ordre croissant :

		      5 ; 2,7 ; -2,6 ; -0,1 ; 7,1 ; -8,3

		      \dotfill
		      % -8,3 < -2,6 < -0,1 < 2,7 < 5 < 7,1
		\item Range les nombres suivants dans l'ordre décroissant :

		      -10,6 ; 14 ; -8,31 ; -3,8 ; 4,2 ; -8,3

		      \dotfill
		      % 14 > 4,2 > -3,8 > -8,3 > -8,31 > -10,6
	\end{enumerate}
\end{question}
